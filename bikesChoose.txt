#include <iostream>
#include <cmath>
#include <string>
using namespace std;

enum enBikes { Hayabusa = 1, QJ, CBR, Harely, MvAgusta, gxss, gxrr };

void BikesMenu()
{
	cout << "====================" << endl;
	cout << "Bikes Menu" << endl;
	cout << "====================" << endl;
	cout << "(1) Hayabusa" << endl;
	cout << "(2) Qj" << endl;
	cout << "(3) CBR" << endl;
	cout << "(4) Harely" << endl;
	cout << "(5) MvAgusta" << endl;
	cout << "(6) Gxss" << endl;
	cout << "(7) Gxrr" << endl;
	cout << "----------------------" << endl;
	cout << "Please select a bike?" << endl;



}
enBikes ReadBike()
{
	int bike;
	cin >> bike;
	while (bike < 1 || bike > 7)
	{
		cout << "Invalid selection! Please choose a valid bike number (1-7): ";
		cin >> bike;
	}
	return (enBikes)bike;
}
string GetBikeData(enBikes bikes)
{
	switch (bikes)
	{
	case enBikes::Hayabusa:
		return "'Hayabusa'\n\nEngine: 1340cc\nPrice: 15K";
		break;
	case enBikes::QJ:
		return "'QJ'\n\nEngine: 1000cc\nPrice: 9K";
		break;
	case enBikes::CBR:
		return "'CBR'\n\nEngine: 600cc\nPrice: 10K";
		break;
	case enBikes::Harely:
		return "'Harely'\n\nEngine: 1800cc\nPrice: 20K";
		break;
	case enBikes::MvAgusta:
		return "'MvAgusta'\n\nEngine: 600cc\nPrice: 8.7K";
		break;
	case enBikes::gxss:
		return "'gxss'\n\nEngine: 750cc\nPrice: 6.5K";
		break;
	case enBikes::gxrr:
		return "'gxrr'\n\nEngine: 600cc\nPrice: 14K";
		break;
	default:
		return "Not Available!";

	}



}



int main()
{

	BikesMenu();
	enBikes Bike = ReadBike();
	cout << "=======================\n\n";
	cout << "Result:\n\n";
	cout << GetBikeData(Bike) << endl;
	cout << "===================" << endl;



	return 0;
}
