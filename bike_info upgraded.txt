#include <iostream>
#include <cmath>
#include <string>
using namespace std;

enum enColor { black = 1, red, blackAndred };

struct stContact
{
	string cell_number;
	string Email;



};

struct stOwnerInfo
{

	string fullname;
	string adress;
	stContact contact;


};

struct stBike
{

	string Brand;
	string Model;
	string year;
	string engine;
	stOwnerInfo owner;
	enColor color;
	int C;


};


void ReadInfo(stBike& info, int id)
{
	cout << "** Bike " << id << " **\n\n";
	cout << "Brand: ";
	cin >> info.Brand;
	cout << "Model: ";
	cin >> info.Model;
	cout << "Engine: ";
	cin >> info.engine;
	cout << "Year: ";
	cin >> info.year;
	cout << "Bike Color (1)Black (2)Red (3)RedAndBlack:  ";
	cin >> info.C;
	cout << "\n\nOwner's information\n\n";
	cout << "Full Name: ";
	cin.ignore(256, '\n');
	getline(cin, info.owner.fullname);
	cout << "Adress: ";
	cin >> info.owner.adress;
	cout << "Phone: ";
	cin >> info.owner.contact.cell_number;
	cout << "E-mail: ";
	cin >> info.owner.contact.Email;

	info.color = (enColor)info.C;
	cout << endl << endl;
}
void PrintInfo(stBike info, int id)
{
	
	cout << "--Bike Information " << id << "\n\n";
	cout << "Brand: " << info.Brand << "  Model: " << info.Model << "  Engine: " << info.engine << "cc " << "  Year: " << info.year;

	switch (info.color)

	{
	case enColor::black:
		cout << "   Color: Black\n\n";
		break;
	case enColor::red:
		cout << "  Color: Red\n\n";
		break;
	case enColor::blackAndred:
		cout << "  Color: Back And Red\n\n";
		break;

	default:
		cout << "  Color: Unknown!\n\n";
	}
	



	

	cout << "--Owner Information:\n\n";
	cout << "Full Name: " << info.owner.fullname << " || Adress: " << info.owner.adress << " || Phone: " << info.owner.contact.cell_number << " || E-mail: " << info.owner.contact.Email << endl << endl;

	cout << "************************\n\n";

}
void ReadAllbikes(stBike bikes[2])
{
	ReadInfo(bikes[0], 1);
	ReadInfo(bikes[1], 2);
	
}
void PrintAllbikes(stBike bikes[2])
{
	PrintInfo(bikes[0], 1);
	PrintInfo(bikes[1], 2);

}


int main()
{

	stBike bike[2];
	
	ReadAllbikes(bike);
	PrintAllbikes(bike);

	


	

	return 0;
}
